    <script>
        const video = document.getElementById('editor-video');
        const videoContainer = document.getElementById('video-container');
        const fileInput = document.getElementById('video-file');
        const imageInput = document.getElementById('image-file');
        const placeholder = document.getElementById('placeholder');
        const playBtn = document.getElementById('play-btn');
        const fileTitle = document.getElementById('file-title');
        const textLayer = document.getElementById('text-layer');
        const imgLayer = document.getElementById('img-layer');
        const cutInfo = document.getElementById('cut-info');
        const layerControl = document.getElementById('layer-control');
        const timelineScrub = document.getElementById('timeline-scrub');
        const currentTimeLbl = document.getElementById('current-time-lbl');
        const totalTimeLbl = document.getElementById('total-time-lbl');

        let startTime = 0;
        let endTime = 0;
        let deleteStart = -1;
        let deleteEnd = -1;
        let activePresets = { pb: false, sepia: false, invert: false };

        // Alternância de Abas da Barra Inferior
        function switchTab(panelId, btnElement) {
            document.querySelectorAll('.tool-panel').forEach(panel => panel.classList.remove('active-panel'));
            document.querySelectorAll('.nav-item').forEach(nav => nav.classList.remove('active-nav'));
            
            document.getElementById(`panel-${panelId}`).classList.add('active-panel');
            btnElement.classList.add('active-nav');
        }

        fileInput.onchange = (e) => {
            const file = e.target.files[0];
            if (file) {
                video.src = URL.createObjectURL(file);
                placeholder.style.display = 'none';
                fileTitle.innerText = file.name;
                
                video.onloadedmetadata = () => {
                    timelineScrub.max = video.duration;
                    startTime = 0;
                    endTime = video.duration;
                    deleteStart = -1;
                    deleteEnd = -1;
                    totalTimeLbl.innerText = formatTime(video.duration);
                    video.play();
                    playBtn.className = "fa-solid fa-pause";
                };
            }
        };

        imageInput.onchange = (e) => {
            const file = e.target.files[0];
            if (file) {
                imgLayer.src = URL.createObjectURL(file);
                imgLayer.style.display = 'block';
                layerControl.style.display = 'flex';
                makeDraggable(imgLayer);
            }
        };

        function formatTime(secs) {
            const m = Math.floor(secs / 60).toString().padStart(2, '0');
            const s = Math.floor(secs % 60).toString().padStart(2, '0');
            return `${m}:${s}`;
        }

        function scrubVideo() {
            if (!video.src) return;
            video.currentTime = parseFloat(timelineScrub.value);
        }

        function splitVideo(type) {
            if (!video.src) return;
            const currentPosition = video.currentTime;
            
            if (type === 'before') {
                startTime = currentPosition;
            } else if (type === 'after') {
                endTime = currentPosition;
            }
            
            updateDisplayInfo();
            video.currentTime = startTime;
        }

        function markSegment(type) {
            if (!video.src) return;
            if (type === 'start') {
                deleteStart = video.currentTime;
            } else if (type === 'end') {
                deleteEnd = video.currentTime;
            }
            updateDisplayInfo();
        }

        function deleteSegment() {
            if (deleteStart !== -1 && deleteEnd !== -1 && deleteEnd > deleteStart) {
                cutInfo.innerText = `Corte do meio ativo`;
                video.currentTime = startTime;
            } else {
                alert("Marque o início e o fim do segmento antes de remover.");
            }
        }

        function updateDisplayInfo() {
            if (deleteStart !== -1 && deleteEnd !== -1) {
                cutInfo.innerText = `Removendo meio`;
